UDE Grade Calculator

Create a modern, responsive web application using only HTML, CSS, and JavaScript that calculates a student’s weighted university grade according to the grading system used at Universität Duisburg-Essen (UDE), Germany. The application should calculate the final grade as a weighted average, where the ECTS credits of each course serve as the weights. The calculation must use the formula: Final Grade = Σ(Grade × ECTS Credits) ÷ Σ(ECTS Credits). The application should allow users to add an unlimited number of courses. Each course should include an optional course name, a grade selected from a dropdown menu, and the number of ECTS credits. Only the following grades should be accepted: 1.0, 1.3, 1.7, 2.0, 2.3, 2.7, 3.0, 3.3, 3.7, and 4.0. Each course should also have a delete button so it can be removed individually. The calculator should automatically recalculate the weighted average whenever a grade or credit value changes. It should display both the total number of ECTS credits and the weighted average grade, rounded to two decimal places. Empty rows should be ignored, and the application should validate all inputs to ensure that credits are positive numbers and that only valid grades can be selected. Design the interface with a clean, modern, and responsive layout that works well on both desktop and mobile devices. Use a light color theme with rounded cards, clear typography, and generous spacing. Do not use Bootstrap or any external CSS or JavaScript libraries. Include a “Reset All” button that clears all entered data. Save all entered courses in Local Storage so that the information remains available after refreshing the page. When the application is opened for the first time, preload it with a few example courses to demonstrate how the calculator works. Also display a short explanation of the weighted average formula so users understand how their final grade is calculated. Structure the project into three separate files: index.html, style.css, and script.js. Write clean, modular, and well-commented JavaScript that is easy to maintain and extend in the future. The finished application should look and behave like a professional grade calculator specifically designed for students of Universität Duisburg-Essen.

    { value: '1.0', label: '1.0 — Sehr gut' },

    { value: '2.0', label: '2.0 — Gut' },

/**

 */
function generateId() {
    return `course-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
}

/**
 * Validates if a string represents a positive number.
 * @param {string} value - The value to validate
 * @returns {boolean} True if valid positive number
 */
function isValidPositiveNumber(value) {
    if (!value || value.trim() === '') return false;
    const num = parseFloat(value);
    return !isNaN(num) && num > 0 && isFinite(num);
}

/**
 * Validates if a grade is in the allowed set.
 * @param {string} grade - The grade value to check
 * @returns {boolean} True if grade is valid
 */
function isValidGrade(grade) {
    return VALID_GRADES.some(g => g.value === grade);
}

/**
 * Determines the German grade classification for a given average.
 * @param {number} average - The calculated weighted average
 * @returns {Object} Object containing cssClass and description text
 */
function getGradeClassification(average) {
    if (!average || isNaN(average)) {
        return { class: 'grade-empty', text: 'Enter courses to see your grade' };
    }
    for (const classification of GRADE_CLASSIFICATIONS) {
        if (average <= classification.max) {
            return { class: classification.class, text: classification.text };
        }
    }
    return { class: 'grade-empty', text: '' };
}

/**
 * Saves current courses array to Local Storage.
 */
function saveToStorage() {
    try {
        localStorage.setItem(STORAGE_KEY, JSON.stringify(courses));
    } catch (e) {
        console.warn('Failed to save to Local Storage:', e);
    }
}

/**
 * Loads courses from Local Storage or returns null if none exist.
 * @returns {Array|null} Parsed courses array or null
 */
function loadFromStorage() {
    try {
        const data = localStorage.getItem(STORAGE_KEY);
        if (data) {
            const parsed = JSON.parse(data);
            if (Array.isArray(parsed)) return parsed;
        }
    } catch (e) {
        console.warn('Failed to load from Local Storage:', e);
    }
    return null;
}

/**
 * Creates the HTML for a grade select dropdown.
 * @param {string} selectedValue - Currently selected grade value
 * @returns {string} HTML string for select element options
 */
function createGradeOptions(selectedValue = '') {
    let options = '<option value="" disabled selected>Grade</option>';
    VALID_GRADES.forEach(g => {
        const selected = g.value === selectedValue ? ' selected' : '';
        options += `<option value="${g.value}"${selected}>${g.label}</option>`;
    });
    return options;
}

/**
 * Creates and returns a DOM element for a single course row.
 * @param {Object} course - Course data object
 * @returns {HTMLElement} The constructed row element
 */
function createCourseRow(course) {
    const row = document.createElement('div');
    row.className = 'course-row';
    row.dataset.id = course.id;

    // Course Name Input
    const nameInput = document.createElement('input');
    nameInput.type = 'text';
    nameInput.name = 'courseName';
    nameInput.placeholder = 'e.g. Linear Algebra';
    nameInput.value = course.name || '';
    nameInput.autocomplete = 'off';
    nameInput.addEventListener('input', () => updateCourse(course.id, 'name', nameInput.value));

    // Grade Select
    const gradeSelect = document.createElement('select');
    gradeSelect.name = 'grade';
    gradeSelect.innerHTML = createGradeOptions(course.grade);
    gradeSelect.addEventListener('change', () => {
        validateInput(gradeSelect, isValidGrade(gradeSelect.value));
        updateCourse(course.id, 'grade', gradeSelect.value);
    });

    // ECTS Input
    const ectsInput = document.createElement('input');
    ectsInput.type = 'number';
    ectsInput.name = 'ects';
    ectsInput.placeholder = 'ECTS';
    ectsInput.min = '0.5';
    ectsInput.step = '0.5';
    ectsInput.value = course.ects || '';
    ectsInput.addEventListener('input', () => {
        validateInput(ectsInput, isValidPositiveNumber(ectsInput.value));
        updateCourse(course.id, 'ects', ectsInput.value);
    });

    // Delete Button
    const deleteBtn = document.createElement('button');
    deleteBtn.type = 'button';
    deleteBtn.className = 'btn-delete';
    deleteBtn.title = 'Remove course';
    deleteBtn.innerHTML = `<svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><polyline points="3 6 5 6 21 6"></polyline><path d="M19 6v14a2 2 0 0 1-2 2H7a2 2 0 0 1-2-2V6m3 0V4a2 2 0 0 1 2-2h4a2 2 0 0 1 2 2v2"></path></svg>`;
    deleteBtn.addEventListener('click', () => removeCourse(course.id));

    // Assemble row
    row.appendChild(nameInput);
    row.appendChild(gradeSelect);
    row.appendChild(ectsInput);
    row.appendChild(deleteBtn);

    // Initial validation visual state
    if (course.grade) validateInput(gradeSelect, isValidGrade(course.grade));
    if (course.ects) validateInput(ectsInput, isValidPositiveNumber(course.ects));

    return row;
}

/**
 * Applies visual validation classes to an input element.
 * @param {HTMLElement} element - The input element to style
 * @param {boolean} isValid - Whether the input is valid
 */
function validateInput(element, isValid) {
    element.classList.remove('valid', 'invalid');
    if (element.value.trim() !== '') {
        element.classList.add(isValid ? 'valid' : 'invalid');
    }
}

/**
 * Adds a new empty course to the state and re-renders.
 */
function addCourse() {
    const newCourse = {
        id: generateId(),
        name: '',
        grade: '',
        ects: ''
    };
    courses.push(newCourse);
    render();
    // Focus the new row's name input for better UX

        if (nameInput) nameInput.focus();
    }

/**
 * Updates a specific field of a course and recalculates.
 * @param {string} id - Course ID
 * @param {string} field - Field name to update
 * @param {string} value - New value
 */
function updateCourse(id, field, value) {
    const course = courses.find(c => c.id === id);
    if (course) {
        course[field] = value;
        saveToStorage();
        calculateAndDisplay();
    }
}

/**
 * Removes a course from the state and re-renders.
 * @param {string} id - Course ID to remove
 */
function removeCourse(id) {
    courses = courses.filter(c => c.id !== id);
    saveToStorage();
    render();
}

/**
 * Resets all data to default example courses.
 */
function resetAll() {
    if (courses.length > 0) {
        // Only confirm if there is actual user data (not just empty rows)
        const hasData = courses.some(c => c.grade || c.ects || c.name);
        if (hasData && !confirm('Are you sure you want to clear all entered courses?')) {
            return;
        }
    }
    courses = DEFAULT_COURSES.map(c => ({ ...c, id: generateId() }));
    saveToStorage();
    render();
}

// ============================================
// Calculation Logic
// ============================================

/**
 * Calculates the weighted average and updates the display.
 * Empty or invalid rows are automatically ignored.
 */
function calculateAndDisplay() {
    let totalWeighted = 0;
    let totalECTS = 0;
    let validCourseCount = 0;

    courses.forEach(course => {
        const grade = parseFloat(course.grade);
        const ects = parseFloat(course.ects);

        // Only include rows with both valid grade and valid positive ECTS
        if (isValidGrade(course.grade) && isValidPositiveNumber(course.ects)) {
            totalWeighted += grade * ects;
            totalECTS += ects;
            validCourseCount++;
        }
    });

    // Update Total ECTS display
    totalECTSEl.textContent = totalECTS > 0 ? totalECTS.toFixed(1) : '0';

    // Calculate and update Final Grade
    let finalGrade = null;
    if (totalECTS > 0) {
        finalGrade = totalWeighted / totalECTS;
        finalGradeEl.textContent = finalGrade.toFixed(2);
    } else {
        finalGradeEl.textContent = '—';
    }

    // Update grade classification description
    const classification = getGradeClassification(finalGrade);
    gradeDescriptionEl.className = `grade-description ${classification.class}`;
    gradeDescriptionEl.textContent = classification.text;
}

/**
 * Renders the entire course list from state.
 */
function render() {
    courseListEl.innerHTML = '';

    if (courses.length === 0) {
        // Show empty state
        courseListEl.innerHTML = `
            <div class="empty-state">
                <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="1.5" stroke-linecap="round" stroke-linejoin="round"><path d="M22 10v6M2 10l10-5 10 5-10 5z"/><path d="M6 12v5c0 2 2 3 6 3s6-1 6-3v-5"/></svg>
                <p>No courses added yet. Click "Add Course" to begin.</p>
            </div>
        `;
    } else {
        courses.forEach(course => {
            courseListEl.appendChild(createCourseRow(course));
        });
    }

    calculateAndDisplay();
}

/**
 * Initializes the application on page load.
 * Loads data from storage or seeds with defaults.
 */
function init() {
    // Load saved data or use defaults for first-time visitors
    const saved = loadFromStorage();
    if (saved && saved.length > 0) {
        courses = saved;
    } else {
        // First visit: populate with examples
        courses = DEFAULT_COURSES.map(c => ({ ...c, id: generateId() }));
        saveToStorage();
    }

    // Bind global button events
    addCourseBtn.addEventListener('click', addCourse);
    resetBtn.addEventListener('click', resetAll);

    // Initial render
    render();
}

if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', init);

    init();
}
